Route::get('/', function () {
    if (Auth::check()) {
        $user = Auth::user();
        if ($user->isAdmin()) {
            return redirect()->route('admin.dashboard');
        }
        if ($user->isProfessor()) {
            return redirect()->route('professor.dashboard');
        }
        if ($user->isStudent()) {
            return redirect()->route('student.dashboard');
        }
        return redirect()->route('auth.login');
    }

    return redirect()->route('login');
});
