<script>
    document.addEventListener('DOMContentLoaded', function () {
        // --- DATA FROM PHP ---
        const rooms = {!! json_encode($rooms->map(fn($r) => ['id' => $r->id, 'room_number' => $r->room_number])) !!};
        const khmerDays = {
            'Monday': 'ច័ន្ទ', 'Tuesday': 'អង្គារ', 'Wednesday': 'ពុធ', 'Thursday': 'ព្រហស្បតិ៍', 
            'Friday': 'សុក្រ', 'Saturday': 'សៅរ៍', 'Sunday': 'អាទិត្យ'
        };
        const initialProgramId = document.getElementById('program_id').value;
        const initialGeneration = document.getElementById('generation').value;
        const selectedCourseId = {{ $courseOffering->course_id }};

        // --- DOM ELEMENTS ---
        const programSelect = document.getElementById('program_id');
        const generationSelect = document.getElementById('generation');
        const courseSelect = document.getElementById('course_id');
        const addScheduleBtn = document.getElementById('add-schedule-btn');
        const schedulesContainer = document.getElementById('schedules-container');

        // --- DEPENDENT DROPDOWN LOGIC ---
        function updateCourses(programId, generation, defaultCourseId = null) {
            courseSelect.innerHTML = '<option value="">{{ __("កំពុងផ្ទុក...") }}</option>';
            courseSelect.disabled = true;

            if (!programId || !generation) {
                courseSelect.innerHTML = '<option value="">{{ __("សូមជ្រើសរើសកម្មវិធីសិក្សានិងជំនាន់សិន") }}</option>';
                return;
            }

            fetch(`/admin/get-courses-by-program-and-generation?program_id=${programId}&generation=${generation}`)
                .then(response => response.json())
                .then(courses => {
                    courseSelect.innerHTML = '<option value="">{{ __("ជ្រើសរើសមុខវិជ្ជា") }}</option>';
                    if (courses.length > 0) {
                        courses.forEach(course => {
                            const option = document.createElement('option');
                            option.value = course.id;
                            option.textContent = `${course.title_km}`;
                            if (course.id == defaultCourseId) {
                                option.selected = true;
                            }
                            courseSelect.appendChild(option);
                        });
                    } else {
                        courseSelect.innerHTML = '<option value="">{{ __("មិនមានមុខវិជ្ជាសម្រាប់កម្មវិធីសិក្សានិងជំនាន់នេះទេ") }}</option>';
                    }
                    courseSelect.disabled = false;
                })
                .catch(error => {
                    console.error('Error fetching courses:', error);
                    courseSelect.innerHTML = '<option value="">{{ __("មានបញ្ហាក្នុងការផ្ទុកមុខវិជ្ជា") }}</option>';
                });
        }

        programSelect.addEventListener('change', () => updateCourses(programSelect.value, generationSelect.value));
        generationSelect.addEventListener('change', () => updateCourses(programSelect.value, generationSelect.value));
        
        // Initial load for the edit page
        if (initialProgramId && initialGeneration) {
            updateCourses(initialProgramId, initialGeneration, selectedCourseId);
        }

        // --- ADD SCHEDULE LOGIC ---
        function createOptions(data, type) {
            let options = type === 'day' ? '<option value="">ជ្រើសរើសថ្ងៃ</option>' : '<option value="">ជ្រើសរើសបន្ទប់</option>';
            if (type === 'day') {
                for (const key in data) { options += `<option value="${key}">${data[key]}</option>`; }
            } else { // room
                data.forEach(item => { options += `<option value="${item.id}">${item.room_number}</option>`; });
            }
            return options;
        }

        function addScheduleField() {
            const uniqueId = Date.now();
            const scheduleItem = document.createElement('div');
            scheduleItem.className = 'schedule-item grid grid-cols-2 lg:grid-cols-4 gap-4 items-end p-4 border rounded-lg bg-gray-50 relative';
            scheduleItem.innerHTML = `
                <button type="button" class="remove-schedule-btn absolute -top-2 -right-2 h-7 w-7 bg-red-500 text-white rounded-full shadow hover:bg-red-600 flex items-center justify-center" onclick="this.closest('.schedule-item').remove()">✕</button>
                <div class="col-span-2 lg:col-span-4">
                    <label class="block text-sm font-medium text-gray-700">ថ្ងៃ</label>
                    <select name="schedules[${uniqueId}][day_of_week]" class="mt-1 block w-full rounded-xl border-gray-300 shadow-sm" required>${createOptions(khmerDays, 'day')}</select>
                </div>
                <div class="col-span-2">
                    <label class="block text-sm font-medium text-gray-700">បន្ទប់</label>
                    <select name="schedules[${uniqueId}][room_id]" class="mt-1 block w-full rounded-xl border-gray-300 shadow-sm" required>${createOptions(rooms, 'room')}</select>
                </div>
                <div>
                    <label class="block text-sm font-medium text-gray-700">ម៉ោងចាប់ផ្តើម</label>
                    <input type="time" name="schedules[${uniqueId}][start_time]" class="mt-1 block w-full rounded-xl border-gray-300 shadow-sm" required>
                </div>
                <div>
                    <label class="block text-sm font-medium text-gray-700">ម៉ោងបញ្ចប់</label>
                    <input type="time" name="schedules[${uniqueId}][end_time]" class="mt-1 block w-full rounded-xl border-gray-300 shadow-sm">
                </div>
            `;
            schedulesContainer.appendChild(scheduleItem);
        }

        addScheduleBtn.addEventListener('click', addScheduleField);

        // Don't add a new field if schedules already exist
        if (schedulesContainer.children.length === 0) {
            addScheduleField();
        }
    });
</script>
